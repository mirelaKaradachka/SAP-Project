create table User (
   id INT NOT NULL auto_increment,
   username VARCHAR(32) NOT NULL,
   password  VARCHAR(32) NOT NULL,
   email  VARCHAR(64) NOT NULL,
   firstname  VARCHAR(32) NOT NULL,
   lastname  VARCHAR(32) NOT NULL,
   age INT  default NOT NULL,
   isMale BIT NOT NULL,
   weight DOUBLE NOT NULL,
   height DOUBLE NOT NULL,
   activityLevel DOUBLE NOT NULL,
   achievementPoints INT NOT NULL,
   workoutPlan VARCHAR(255) NOT NULL,
   PRIMARY KEY (id)
);

create table Day (
   id INT NOT NULL auto_increment,
   name VARCHAR(32) NOT NULL,
   comparison DOUBLE NOT NULL,
   PRIMARY KEY (id)
);

create table User_Day (
   user_id INT NOT NULL,
   day_id INT NOT NULL,
   PRIMARY KEY (user_id,day_id)
);

create table Meal (
   id INT NOT NULL auto_increment,
   name VARCHAR(32) NOT NULL,
   PRIMARY KEY (id)
);

create table Day_Meal (
   days_id INT NOT NULL,
   meals_id INT NOT NULL,
   PRIMARY KEY (days_id,meals_id)
);

create table Course (
   id INT NOT NULL auto_increment,
   meal_ids DEFAULT NULL,
   name VARCHAR(32) NOT NULL,
   calories DOUBLE NOT NULL,
   proteins DOUBLE NOT NULL,
   fats DOUBLE NOT NULL,
   carbohydrates DOUBLE NOT NULL,
   PRIMARY KEY (id),
   FOREIGN KEY (meal_ids) REFERENCES Meal(id)
);
